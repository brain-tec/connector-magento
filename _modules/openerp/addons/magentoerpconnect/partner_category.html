    [docs]    @only_create
    @mapping
    def openerp_id(self, record):
        """ Will bind the category on a existing one with the same name."""
        existing = self.env['res.partner.category'].search(
            [('name', '=', record['customer_group_code'])],
            limit=1,
        )
        if existing:
            return {'openerp_id': existing.id}
